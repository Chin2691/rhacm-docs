apiVersion: v1

  name: mc-subscription-metrics
  namespace: open-cluster-management-agent-addon
spec:
  ports:
  - name: metrics
    port: 8388
    protocol: TCP
    targetPort: 8388
  selector:
    component: application-manager
  sessionAffinity: None
  type: ClusterIP
